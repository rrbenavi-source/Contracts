4. Power Queries (M Language) – 8 Consultas

Arquitectura: PQ-01 (Contratos_Base) contiene la única conexión a Databricks. PQ-02 a PQ-08 referencian Contratos_Base con = Contratos_Base. Cada query es una "Consulta en blanco" separada en Power BI.
PQ-01 · Contratos_Base · Conexión principal + columnas calculadasDatabricks → tabla base filtrada
let
    // ── Conexión directa a Databricks (valores literales, sin parámetros) ──
    Origen = Databricks.Catalogs(
        "adb-1347980599310379.19.azuredatabricks.net",
        "/sql/1.0/warehouses/d956a286a35ef0b4",
        [Catalog = "heiaepmx002pwe01"]
    ),
    Catalogo     = Origen{[Name="heiaepmx002pwe01"]}[Data],
    Schema       = Catalogo{[Name="gold_hnk_mx_loc_procure"]}[Data],
    Tabla        = Schema{[Name="tbl_sot_contracts"]}[Data],

    // ── Materializar en memoria para romper query folding ──
    TablaBuf     = Table.Buffer(Tabla),

    // ── Filtrar solo contratos activos (protege nulls) ──
    Activos      = Table.SelectRows(TablaBuf,
                       each Value.Equals([contrato_activo], true)),

    // ── Castear tipos base ──
    Tipado       = Table.TransformColumnTypes(Activos, {
                       {"periodo_validez_inicio", type date},
                       {"dias_de_credito",         Int64.Type},
                       {"contrato_activo",          type logical}
                   }),

    // ── Calcular Anio, Mes y AnioMesFecha desde periodo_validez_inicio ──
    ConAnio      = Table.AddColumn(Tipado, "anio", each
                       if [periodo_validez_inicio] = null then null
                       else Date.Year([periodo_validez_inicio]),
                       Int64.Type),

    ConMes       = Table.AddColumn(ConAnio, "mes", each
                       if [periodo_validez_inicio] = null then null
                       else Date.Month([periodo_validez_inicio]),
                       Int64.Type),

    ConFecha     = Table.AddColumn(ConMes, "anio_mes_fecha", each
                       if [periodo_validez_inicio] = null then null
                       else #date(Date.Year([periodo_validez_inicio]),
                                  Date.Month([periodo_validez_inicio]), 1),
                       type date),

    // ── Agregar descripción del tipo de contrato ──
    ConTipo      = Table.AddColumn(ConFecha, "Tipo_Contrato_Desc", each
                       if [documentos_compras_clase] = "YELO" then "Local (YELO)"
                       else if [documentos_compras_clase] = "YECL" then "Global (YECL)"
                       else [documentos_compras_clase]),

    // ── Agregar rango de días de crédito ──
    ConRango     = Table.AddColumn(ConTipo, "Rango_Dias_Credito", each
                       if [dias_de_credito] = null  then "Sin dato"
                       else if [dias_de_credito] = 0 then "0 días"
                       else if [dias_de_credito] <= 30 then "1-30 días"
                       else if [dias_de_credito] <= 60 then "31-60 días"
                       else if [dias_de_credito] <= 90 then "61-90 días"
                       else if [dias_de_credito] <= 120 then "91-120 días"
                       else if [dias_de_credito] <= 180 then "121-180 días"
                       else "180+ días"),

    // ── Columna de anomalía ──
    ConAnomalia  = Table.AddColumn(ConRango, "es_anomalia_credito", each
                       [dias_de_credito] = 0 and
                       not Value.Equals([condiciones_pago_clave], "V000")),

    // ── Seleccionar columnas finales ──
    Final        = Table.SelectColumns(ConAnomalia, {
                       "documento_compras_num", "documentos_compras_clase",
                       "Tipo_Contrato_Desc", "condiciones_pago_clave",
                       "dias_de_credito", "Rango_Dias_Credito",
                       "contrato_activo", "es_anomalia_credito",
                       "periodo_validez_inicio", "periodo_validez_fin",
                       "anio", "mes", "anio_mes_fecha",
                       "proveedor_id", "proveedor_nombre",
                       "moneda_clave", "importe_neto"
                   })
in
    Final
PQ-02 · KPI_Contratos_Generales · Tabla de 6 KPIsReferencia Contratos_Base
let
    Base = Contratos_Base,

    // ── KPI 1: Total contratos activos únicos ──
    TotalContratos     = Table.RowCount(Table.Distinct(Base, {"documento_compras_num"})),

    // ── KPI 2: Contratos YELO (Locales) ──
    ContratosYELO      = Table.RowCount(Table.Distinct(
                             Table.SelectRows(Base, each [documentos_compras_clase] = "YELO"),
                             {"documento_compras_num"})),

    // ── KPI 3: Contratos YECL (Globales) ──
    ContratosYECL      = Table.RowCount(Table.Distinct(
                             Table.SelectRows(Base, each [documentos_compras_clase] = "YECL"),
                             {"documento_compras_num"})),

    // ── KPI 4: Promedio días de crédito ──
    PromedioDias       = List.Average(Table.Distinct(Base, {"documento_compras_num"})[dias_de_credito]),

    // ── KPI 5: Contratos con 0 días y condición ≠ V000 (anomalías) ──
    Anomalias          = Table.RowCount(Table.Distinct(
                             Table.SelectRows(Base, each
                                 [dias_de_credito] = 0 and
                                 not Value.Equals([condiciones_pago_clave], "V000")),
                             {"documento_compras_num"})),

    // ── KPI 6: % contratos YELO ──
    PctYELO            = if TotalContratos = 0 then 0
                         else Number.Round(ContratosYELO / TotalContratos * 100, 1),

    // ── Tabla de resultados ──
    Resultado = Table.FromRows({
        {"Total_Contratos_Activos",    TotalContratos},
        {"Contratos_YELO_Local",       ContratosYELO},
        {"Contratos_YECL_Global",      ContratosYECL},
        {"Promedio_Dias_Credito",       Number.Round(PromedioDias, 0)},
        {"Anomalias_Credito_Cero",     Anomalias},
        {"Pct_YELO",                   PctYELO}
    }, {"KPI", "Valor"})
in
    Resultado

PQ-08 · Matriz_Condicion_Tipo · Pivot YELO / YECL por condición de pagoTabla cruzada
let
    Base       = Contratos_Base,

    // ── Materializar para evitar errores de null en pivot ──
    BaseBuf    = Table.Buffer(Base),

    Distintos  = Table.Distinct(BaseBuf, {"documento_compras_num"}),

    // ── Contar por condición + tipo ──
    Agrupado   = Table.Group(Distintos,
                     {"condiciones_pago_clave", "documentos_compras_clase"}, {
                         {"Contratos", each Table.RowCount(_), Int64.Type}
                 }),

    // ── Pivot: columnas = YELO, YECL ──
    Pivoteado  = Table.Pivot(
                     Agrupado,
                     List.Distinct(Agrupado[documentos_compras_clase]),
                     "documentos_compras_clase",
                     "Contratos",
                     List.Sum
                 ),

    // ── Reemplazar nulls por 0 ──
    Limpio     = Table.ReplaceValue(Pivoteado, null, 0,
                     Replacer.ReplaceValue, Table.ColumnNames(Pivoteado)),

    // ── Agregar total por fila ──
    ConTotal   = Table.AddColumn(Limpio, "Total", each
                     (if Record.HasFields(_, "YELO") then _[YELO] else 0) +
                     (if Record.HasFields(_, "YECL") then _[YECL] else 0),
                     Int64.Type),

    Ordenado   = Table.Sort(ConTotal, {{"Total", Order.Descending}})
in
    Ordenado

Reglas críticas del modelo:

• Todas las tablas de agregados usan Table.Distinct(_, {"documento_compras_num"}) — NUNCA contar registros directamente.
• Contratos_Base usa Table.Buffer() para romper query folding antes del filtro de contrato_activo.
• El filtro de contrato activo usa Value.Equals([contrato_activo], true) para proteger nulos.
• Los campos anio, mes, anio_mes_fecha son calculados en PQ-01, NO leídos de la vista (devuelve 0).
• NO hay relaciones activas entre tablas en el modelo (tablas independientes pre-agregadas).
• Los KPI Cards se configuran con filtro visual: campo KPI = nombre del KPI correspondiente.
